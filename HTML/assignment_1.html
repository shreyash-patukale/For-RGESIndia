BASIC FORMULAE IN ALGEBRA

(a + b)3 = a3 + b3 + 3a2b + 3ab2

(a - b)3 = a3 - b3 - 3a2b + 3ab2

a3 + b3 = (a + b)(a2 - ab + b2)

a3 - b3 = (a - b)(a2 + ab + b2)
